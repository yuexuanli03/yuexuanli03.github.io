# make a copy of data and group the data by continent and then make a scatter plot per continent 
copy1 = data[['continent','year','lifeExp']]
continent_list = data['continent'].unique()
copy1 = copy1.groupby(['continent'])

for c in continent_list:
    exp = copy1.get_group(c)
    x = exp['year']
    y = exp['lifeExp']
    plt.figure(figsize = (5,5))
    
    # make a scatter plot
    plt.scatter(x,y)
    plt.title("Life expectancy vs. year")
    plt.xlabel('year')
    plt.ylabel('life expectancy')

    # plot regression line: use np.polyfit to find coefficients and then plot it
    coefficients = np.polyfit(x,y,1)
    plt.plot(x, coefficients[0]*x + coefficients[1], c = 'r')
    plt.show()

from sklearn import neighbors, datasets

knn = neighbors.KNeighborsClassifier(n_neighbors = 1)

# use it to predict labels for data we already know
knn.fit(X, y)
y_model = knn.predict(X)

# compute the fraction of correctly labeled points
accuracy_score(y, y_model)

Here, I got an accuracy score of 1.0 which means that 100% of points were correctly labeled by the model. However, this approach contains a flaw that is it trains and evaluates the model on the same data. So, I want to use holdout set to redo the analyze.

Now perform holdout validation on knn classifier: holdout sets: hold back some subset of the data from the training of the model, and then use this holdout set to check the model performance

In [49]:
